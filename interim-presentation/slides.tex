    upper :: String -> String
    upper []       = []
    upper (x : xs) =
      toUpper x : upper xs
    \end{lstlisting}
\end{frame}

\begin{frame}[fragile]{TEMPLATE}
    \begin{lstlisting}
    evens :: [Int] -> [Int]
    evens []       = []
    evens (x : xs)
      | even x     = x : evens xs
      | otherwise  = evens xs
    \end{lstlisting}
\end{frame}

\begin{frame}[fragile]{TEMPLATE}

    sum :: [Int] -> Int
    sum []       = 0
